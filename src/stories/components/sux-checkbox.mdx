# Checkbox

Buttons allow users to perform an action or to navigate to another page. They have multiple styles for various needs, and are ideal for calling attention to where a user needs to do something in order to move forward in a flow.

export const Defaultcheckbox = args => {
  return html`
    <div style="padding: 10%; display: flex; justify-content: center; gap:16px;">
      <sux-checkbox .label="${args.label}" variant="info" .size="${args.size}"> Tag info </sux-checkbox>
      <sux-checkbox .label="${args.label}" variant="success" .size="${args.size}"> Tag success </sux-checkbox>
      <sux-checkbox .label="${args.label}" variant="warning" .size="${args.size}"> Tag warning </sux-checkbox>
      <sux-checkbox .label="${args.label}" variant="error" .size="${args.size}"> Tag error </sux-checkbox>
      <sux-checkbox .label="${args.label}" variant="neutral" .size="${args.size}"> Tag neutral </sux-checkbox>
    </div>
  `;
};

<Canvas>
  <Story name="Playground">{Defaultcheckbox.bind()}</Story>
</Canvas>
